```

    folders:{
    	permissionType:'连网透传',
    	roleId:'连网透传',
        '之前的数据'
    }
    permissions:{
        check:[{
            serNodeNamePath:'外域路径',
            folderPath:'目录路径',
            deviceId:null,
            permissionType:'VOD|RECORD_LOCAL|RECORD_SERVER|DOWNLOAD|PTZ'（这个地方会根据传来的权限类型返回）,
            sourceType:'FOLDER',
            roleId:'角色id'
        }]
```
